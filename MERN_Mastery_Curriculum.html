JavaScript Fundamentals

The foundation everything else is built on — master these first

1
Closures, Scope & the Execution Context
HardConcept
▼
What is a Closure?
A closure is a function that remembers variables from its outer scope even after the outer function has finished executing. JavaScript creates a closure every time a function is created.
function makeCounter() {
  let count = 0; // this variable is "closed over"
  return function() {
    count++;
    return count;
  };
}

const counter = makeCounter();
console.log(counter()); // 1
console.log(counter()); // 2
// count is not accessible from outside, but the inner function remembers it
Real Use Cases
→Data privacy — hide implementation details (like private variables)
→Factory functions — create customized functions
→Memoization — cache results inside a closure
→Event handlers — retain access to outer state
Classic interview trap: using var in loops. var is function-scoped so all callbacks share the same variable. Use let (block-scoped) or an IIFE to fix.
// ❌ All print 3
for (var i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 100);
}

// ✅ Prints 0, 1, 2
for (let i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 100);
}

Promises, async/await & Error Handling
HardPractical
▼
Promise States
A Promise has 3 states: pending, fulfilled, or rejected. Once settled it never changes.
// Promise chaining
fetch('/api/user')
  .then(res => res.json())
  .then(user => fetch(`/api/posts/${user.id}`))
  .then(res => res.json())
  .catch(err => console.error(err)); // catches ANY error above

// async/await — same thing, cleaner
async function getUserPosts() {
  try {
    const res = await fetch('/api/user');
    const user = await res.json();
    const posts = await fetch(`/api/posts/${user.id}`);
    return await posts.json();
  } catch (err) {
    console.error(err);
  }
}
Promise.all vs Promise.allSettled vs Promise.race
// Promise.all — runs in PARALLEL, fails if ANY fails
const [users, posts] = await Promise.all([getUsers(), getPosts()]);

// Promise.allSettled — runs all, never throws, gives status of each
const results = await Promise.allSettled([getUsers(), getPosts()]);
results.forEach(r => {
  if (r.status === 'fulfilled') console.log(r.value);
  else console.error(r.reason);
});

// Promise.race — resolves/rejects as soon as FIRST settles
const result = await Promise.race([fetchData(), timeout(5000)]);
await only works inside async functions. Forgetting await means you get a Promise object, not the value.
Use Promise.all for parallel independent calls. Never chain sequential awaits when calls don't depend on each other — it wastes time.
3
Event Loop, Call Stack & Microtasks
HardConcept
▼
How it works
JavaScript is single-threaded. The event loop continuously checks if the call stack is empty, then pushes the next callback from the queue. Microtasks (Promises) run before macrotasks (setTimeout).
console.log('1');

setTimeout(() => console.log('2'), 0);

Promise.resolve().then(() => console.log('3'));

console.log('4');

// Output: 1, 4, 3, 2
// Sync first → microtasks (Promise) → macrotasks (setTimeout)
This is why Promise callbacks always run before setTimeout even with 0ms delay — they go in the microtask queue which has higher priority.

Prototypes, this & ES6 Classes
MediumConcept
▼
Prototype Chain
Every JS object has a prototype. When you access a property, JS walks up the prototype chain until it finds it or hits null.
// ES6 Class (syntactic sugar over prototypes)
class Animal {
  constructor(name) { this.name = name; }
  speak() { return `${this.name} makes a noise.`; }
}

class Dog extends Animal {
  speak() { return `${this.name} barks.`; }
}

const d = new Dog('Rex');
console.log(d.speak()); // Rex barks.
console.log(d instanceof Animal); // true
The 'this' keyword — 4 rules
// 1. Implicit binding — left of the dot
obj.method() // this = obj

// 2. Explicit binding
fn.call(obj, arg1)
fn.apply(obj, [args])
const bound = fn.bind(obj)

// 3. new binding
const inst = new Constructor() // this = new object

// 4. Arrow functions — inherit 'this' from enclosing scope
const obj = {
  name: 'test',
  fn: function() {
    const arrow = () => console.log(this.name); // this = obj ✅
    arrow();
  }
};

Debounce, Throttle & Memoization
MediumPractical
▼
Debounce — wait until user stops
function debounce(fn, delay) {
  let timer;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

// React usage — search input
const handleSearch = useCallback(
  debounce((val) => fetchResults(val), 500),
  []
);
// API only called 500ms after user stops typing
Throttle — at most once per interval
function throttle(fn, limit) {
  let lastCall = 0;
  return function(...args) {
    const now = Date.now();
    if (now - lastCall >= limit) {
      lastCall = now;
      return fn.apply(this, args);
    }
  };
}
// Use for: scroll events, resize events, button spam prevention
FeatureDebounceThrottle
FiresAfter user STOPS for N msAt most once every N ms
Use caseSearch input, form validationScroll, resize, mouse move
Guarantees timing?No — resets on each callYes — consistent rate

Hard
▼

Custom Hooks — Build Reusable Logic
MediumPractical
▼
// useFetch — reusable data fetching
function useFetch(url) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    let cancelled = false;
    setLoading(true);
    fetch(url)
      .then(r => r.json())
      .then(d => { if (!cancelled) setData(d); })
      .catch(e => { if (!cancelled) setError(e); })
      .finally(() => { if (!cancelled) setLoading(false); });
    return () => { cancelled = true; }; // cleanup prevents state update on unmount
  }, [url]);

  return { data, loading, error };
}

// useDebounce
function useDebounce(value, delay) {
  const [debounced, setDebounced] = useState(value);
  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);
  return debounced;
}

Error Boundaries & Suspense
Medium
▼
// Error Boundary — must be a class component
class ErrorBoundary extends React.Component {
  state = { hasError: false };

  static getDerivedStateFromError(error) {
    return { hasError: true };
  }

  componentDidCatch(error, info) {
    logErrorToService(error, info); // send to Sentry etc.
  }

  render() {
    if (this.state.hasError) return <h1>Something went wrong.</h1>;
    return this.props.children;
  }
}

2
Streams — Readable, Writable, Transform
HardPractical
▼
Streams process data in chunks. Instead of loading a 1GB file into memory, you process it piece by piece.
const fs = require('fs');
const zlib = require('zlib');

// Without streams — loads ENTIRE file into memory ❌
const data = fs.readFileSync('huge.csv'); // 1GB in RAM!

// With streams — memory efficient ✅
fs.createReadStream('huge.csv')
  .pipe(zlib.createGzip())          // transform: compress
  .pipe(fs.createWriteStream('huge.csv.gz'));  // write

// HTTP response streaming
app.get('/file', (req, res) => {
  res.setHeader('Content-Type', 'text/csv');
  fs.createReadStream('data.csv').pipe(res);
});
Backpressure: if consumer is slower than producer, use stream.pause() and resume on 'drain' event. Node's pipe() handles this automatically.

Clustering & PM2 for Production
Hard
▼
// Node is single-threaded — cluster uses all CPU cores
const cluster = require('cluster');
const os = require('os');

if (cluster.isMaster) {
  const cpuCount = os.cpus().length; // e.g. 8 cores
  for (let i = 0; i < cpuCount; i++) {
    cluster.fork(); // spawn 8 workers
  }
  cluster.on('exit', (worker) => {
    cluster.fork(); // auto-restart crashed worker
  });
} else {
  require('./server'); // each worker runs the server
}

// PM2 (easier, production-ready)
// pm2 start app.js -i max   → auto-detect CPU count
// pm2 restart app           → zero-downtime reload
// pm2 logs / pm2 monit      → monitoring
Workers don't share memory. Don't use in-memory sessions — use Redis instead.

Middleware Chain — Auth & Permissions
Hard
▼

// JWT middleware
function verifyJWT(req, res, next) {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'No token' });
  try {
    req.user = jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch {
    res.status(401).json({ error: 'Invalid token' });
  }
}

// Role middleware
function checkRole(...roles) {
  return (req, res, next) => {
    if (!roles.includes(req.user.role))
      return res.status(403).json({ error: 'Forbidden' });
    next();
  };
}

MongoDB

Schema design, aggregation, indexing and transactions

1
Schema Design — Embed vs Reference
Hard
▼
StrategyWhen to UseExample
EmbedData always accessed together, 1:few relationship, sub-doc rarely changes independentlyUser + addresses, Post + tags
ReferenceData accessed independently, 1:many or many:many, sub-doc updated frequently, large arraysPost + comments (millions), User + orders
// Embedded (good for small, co-accessed data)
{
  _id: ObjectId,
  name: "John",
  addresses: [           // ✅ always fetched with user
    { city: "Chennai", zip: "600001" }
  ]
}

// Referenced (good for large, independent data)
// Post document
{ _id: ObjectId("post1"), title: "My Post", author: ObjectId("user1") }

// Fetch with $lookup (like SQL JOIN)
db.posts.aggregate([
  { $lookup: {
    from: "users",
    localField: "author",
    foreignField: "_id",
    as: "authorDetails"
  }}
]);
MongoDB document size limit is 16MB. Never embed unbounded arrays (e.g., comments on a viral post). Use references + pagination instead.
2
Aggregation Pipeline — Full Guide
Hard
▼
// Top 5 products by revenue per category
db.orders.aggregate([
  // 1. Filter
  { $match: { status: "completed" } },

  // 2. Flatten array of items
  { $unwind: "$items" },

  // 3. Join with products collection
  { $lookup: {
    from: "products",
    localField: "items.productId",
    foreignField: "_id",
    as: "product"
  }},
  { $unwind: "$product" },

  // 4. Group by category + product
  { $group: {
    _id: { category: "$product.category", name: "$product.name" },
    totalRevenue: { $sum: { $multiply: ["$items.qty", "$items.price"] } }
  }},

  // 5. Sort highest first
  { $sort: { totalRevenue: -1 } },

  // 6. Group by category, take top 5
  { $group: {
    _id: "$_id.category",
    topProducts: { $push: { name: "$_id.name", revenue: "$totalRevenue" } }
  }},
  { $project: { topProducts: { $slice: ["$topProducts", 5] } } }
]);
Key stages to memorize: $match (filter), $group (aggregate), $sort, $limit, $lookup (join), $unwind (flatten array), $project (reshape), $facet (multiple pipelines in one).

JWT — Access & Refresh Token Strategy

// Login — issue both tokens
async function login(req, res) {
  const user = await User.findByEmail(req.body.email);
  if (!user || !await user.comparePassword(req.body.password))
    throw new AppError('Invalid credentials', 401);

  const accessToken = jwt.sign(
    { id: user._id, role: user.role },
    process.env.JWT_SECRET,
    { expiresIn: '15m' }  // short-lived
  );

  const refreshToken = jwt.sign(
    { id: user._id },
    process.env.REFRESH_SECRET,
    { expiresIn: '7d' }   // long-lived
  );

  // Store refresh token in httpOnly cookie (XSS safe)
  res.cookie('refreshToken', refreshToken, {
    httpOnly: true, secure: true, sameSite: 'strict',
    maxAge: 7 * 24 * 60 * 60 * 1000
  });

  res.json({ accessToken });
}

// Refresh endpoint
async function refresh(req, res) {
  const token = req.cookies.refreshToken;
  if (!token) throw new AppError('No refresh token', 401);
  const decoded = jwt.verify(token, process.env.REFRESH_SECRET);
  const newAccessToken = jwt.sign(
    { id: decoded.id },
    process.env.JWT_SECRET,
    { expiresIn: '15m' }
  );
  res.json({ accessToken: newAccessToken });
}
Never store access tokens in localStorage (XSS vulnerable). Store in memory (JS variable). Store refresh tokens in httpOnly cookies.

  const cacheKey = `user:${id}`;

// Two Pointer — O(n) instead of O(n²)
function twoSum(arr, target) {
  let left = 0, right = arr.length - 1;
  while (left < right) {
    const sum = arr[left] + arr[right];
    if (sum === target) return [left, right];
    else if (sum < target) left++;
    else right--;
  }
}

// Sliding Window — max sum subarray of size k
function maxSumSubarray(arr, k) {
  let sum = arr.slice(0, k).reduce((a, b) => a + b, 0);
  let max = sum;
  for (let i = k; i < arr.length; i++) {
    sum += arr[i] - arr[i - k]; // slide: add new, remove old
    max = Math.max(max, sum);
  }
  return max;
}

// Frequency Map — O(n) duplicate/count problems
function firstNonRepeating(str) {
  const freq = {};
  for (const c of str) freq[c] = (freq[c] || 0) + 1;
  return str.split('').find(c => freq[c] === 1) || null;
}

// Flatten nested object (common in MERN — API data)
function flatten(obj, prefix = '', result = {}) {
  for (const [key, val] of Object.entries(obj)) {
    const newKey = prefix ? `${prefix}.${key}` : key;
    if (typeof val === 'object' && val !== null && !Array.isArray(val)) {
      flatten(val, newKey, result);
    } else {
      result[newKey] = val;
    }
  }
  return result;
}

flatten({ a: { b: { c: 1 } }, d: 2 });
// → { 'a.b.c': 1, 'd': 2 }

// Deep clone without JSON.parse (handles functions, dates)
function deepClone(obj) {
  if (obj === null || typeof obj !== 'object') return obj;
  if (obj instanceof Date) return new Date(obj);
  if (Array.isArray(obj)) return obj.map(deepClone);
  return Object.fromEntries(
    Object.entries(obj).map(([k, v]) => [k, deepClone(v)])
  );
}

// Debounce
function debounce(fn, delay) {
  let timer = null;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
      timer = null;
    }, delay);
  };
}

// Throttle
function throttle(fn, limit) {
  let inThrottle = false;
  return function(...args) {
    if (!inThrottle) {
      fn.apply(this, args);
      inThrottle = true;
      setTimeout(() => inThrottle = false, limit);
    }
  };
}

// Memoize
function memoize(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) return cache.get(key);
    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

// Binary search — O(log n) on sorted arrays
function binarySearch(arr, target) {
  let left = 0, right = arr.length - 1;
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (arr[mid] === target) return mid;
    else if (arr[mid] < target) left = mid + 1;
    else right = mid - 1;
  }
  return -1;
}

Implement a Rate Limiter from Scratch
Hard

// Sliding Window Rate Limiter (in-memory)
class RateLimiter {
  constructor(maxRequests, windowMs) {
    this.maxRequests = maxRequests;
    this.windowMs = windowMs;
    this.clients = new Map(); // ip → [timestamps]
  }

  isAllowed(clientId) {
    const now = Date.now();
    const windowStart = now - this.windowMs;

    if (!this.clients.has(clientId)) {
      this.clients.set(clientId, []);
    }

    // Remove timestamps outside the window
    const requests = this.clients.get(clientId)
      .filter(ts => ts > windowStart);

    if (requests.length >= this.maxRequests) {
      this.clients.set(clientId, requests);
      return false; // rate limited
    }

    requests.push(now);
    this.clients.set(clientId, requests);
    return true;
  }
}

const limiter = new RateLimiter(100, 60 * 1000); // 100 req/min
app.use((req, res, next) => {
  if (!limiter.isAllowed(req.ip))
    return res.status(429).json({ error: 'Too many requests' });
  next();
});
